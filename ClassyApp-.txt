CREATE TABLE Users (
	Id INT UNSIGNED AUTO_INCREMENT,
	Name VARCHAR(30) NOT NULL,
	Email VARCHAR(50) NOT NULL UNIQUE,
	Password VARCHAR(30) NOT NULL,
	CONSTRAINT Users_PK PRIMARY KEY (Id)
);

CREATE TABLE Classrooms (
	Id INT UNSIGNED,
	Title VARCHAR(30) NOT NULL,
	Description VARCHAR(100),
	OwnerId INT UNSIGNED NOT NULL,
	CONSTRAINT Classrooms_PK PRIMARY KEY (Id),
	CONSTRAINT Classrooms_OwnerId_FK FOREIGN KEY (OwnerId) REFERENCES Users (Id)
		ON DELETE CASCADE
		ON UPDATE CASCADE
);

CREATE TABLE R_UserClassrooms (
	UserId INT UNSIGNED NOT NULL,
	ClassroomId INT UNSIGNED NOT NULL,
	CONSTRAINT R_UserClassrooms_PK PRIMARY KEY (UserId, ClassroomId),
	CONSTRAINT R_UserClassrooms_UserId_FK FOREIGN KEY (UserId) REFERENCES Users (Id)
		ON DELETE CASCADE
		ON UPDATE CASCADE,
	CONSTRAINT R_UserClassrooms_ClassroomId_FK FOREIGN KEY (ClassroomId) REFERENCES Classrooms (Id)
		ON DELETE CASCADE
		ON UPDATE CASCADE
);
